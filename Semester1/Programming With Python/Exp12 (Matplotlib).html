# PLOTTING X and Y POINTS - The plot() function is used to draw points (markers) in a diagram. 
    # By default, the plot() function draws a line from point to point. 

# example1- plot a line from (1, 3) to (8, 10), we have to pass two arrays [1, 8] and [3, 10] to the plot function.

import matplotlib.pyplot as plt
import numpy as np

x_points = np.array([1, 2, 6, 8])    # Parameter 1 is an array containing the points on the x-axis. 
y_points = np.array([3, 8, 1, 10])  # Parameter 2 is an array containing the points on the y-axis. 

plt.plot(x_points,y_points,'c',marker='*',ms=20,linewidth='2',mfc='g',linestyle='dotted')
plt.show()

# PIE CHARTS

import matplotlib.pyplot as plt 
import numpy as np 
y = np.array([35, 25, 25, 15]) 
mylabels = ["Apples", "Bananas", "Guavas", "Cherries"] 
myexplode = [0.2, 0, 0, 0]      # Pull the "Apples" wedge 0.2 from the center of the pie
mycolors = ["black", "hotpink", "c", "#4CAF50"] 
plt.pie(y, labels = mylabels,startangle=90,colors=mycolors,explode=myexplode)
plt.legend(title = "Fruits:") 
plt.show()
